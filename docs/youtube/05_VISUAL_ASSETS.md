## 収束グラフアニメーション（8:00-8:30）

### Python (matplotlib) コード

```python
import matplotlib.pyplot as plt
import matplotlib.animation as animation
import numpy as np

# 設定
fig, ax = plt.subplots(figsize=(16, 9))
fig.patch.set_facecolor('#0a1929')
ax.set_facecolor('#0a1929')

# データ
iterations = np.arange(0, 9)
distances = [1.0, 0.88, 0.75, 0.60, 0.42, 0.29, 0.15, 0.05, 0.0]
convergence = [0, 12, 25, 40, 58, 71, 85, 95, 100]

# グラフスタイル
ax.spines['bottom'].set_color('white')
ax.spines['left'].set_color('white')
ax.spines['top'].set_visible(False)
ax.spines['right'].set_visible(False)
ax.tick_params(colors='white')

# ラベル
ax.set_xlabel('Blinks (Iterations)', color='white', fontsize=20)
ax.set_ylabel('Distance to Goal', color='cyan', fontsize=20)
ax.set_ylabel('Convergence %', color='gold', fontsize=20, rotation=270,
              labelpad=20)

# アニメーション関数
def animate(i):
    ax.clear()

    # 距離のグラフ（青）
    ax.plot(iterations[:i+1], distances[:i+1],
            color='cyan', linewidth=4, marker='o',
            markersize=12, label='Distance to Goal')

    # 収束率のグラフ（金）
    ax2 = ax.twinx()
    ax2.plot(iterations[:i+1], convergence[:i+1],
             color='gold', linewidth=4, marker='s',
             markersize=12, label='Convergence %')

    # 目標線（赤い破線）
    ax.axhline(y=0, color='red', linestyle='--',
               linewidth=2, label='Goal (W_∞)')

    # 現在位置の強調
    if i > 0:
        ax.scatter(iterations[i], distances[i],
                  color='cyan', s=500, zorder=5,
                  edgecolors='white', linewidths=3)

        # テキスト表示
        ax.text(iterations[i], distances[i] + 0.1,
               f'W_{i}\n{convergence[i]}%',
               color='white', fontsize=16,
               ha='center', va='bottom',
               bbox=dict(boxstyle='round',
                        facecolor='black', alpha=0.7))

    ax.set_xlim(-0.5, 8.5)
    ax.set_ylim(-0.1, 1.1)
    ax.legend(loc='upper right')

    return ax,

# アニメーション作成
anim = animation.FuncAnimation(fig, animate, frames=9,
                              interval=500, repeat=True)

# 保存
anim.save('convergence_animation.mp4', writer='ffmpeg',
         fps=2, dpi=150)

plt.show()
```

**出力**: `convergence_animation.mp4`（16:9、1920×1080、5秒）
